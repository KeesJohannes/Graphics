Driving force: recursive generator function

function* CutSierpinsky(pl,pr,pt,n) {

  let plr = halfbetween(pl,pr);

  let prt = halfbetween(pr,pt);

  let ptl = halfbetween(pt,pl);

  yield n; // 

  DrawTriangle(plr,prt,ptl);

  if (n>1) {

    yield* CutSierpinsky(pl,plr,ptl,n-1);

    yield* CutSierpinsky(plr,pr,prt,n-1);

    yield* CutSierpinsky(ptl,prt,pt,n-1);

  }

}
